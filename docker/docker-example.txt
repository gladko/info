//////////////////////////////////////////////////////////
		SWARM
//////////////////////////////////////////////////////////
docker swarm leave --force				# Take down the swarm.
docker node ls						# View the nodes in this swarm

docker swarm init --advertise-addr <swarm-manager-ip>	# Init swarm master

docker swarm join --token <token> <swarm-manager-ip>:2377

docker swarm join-token worker				# Shows command for joining to swarm
